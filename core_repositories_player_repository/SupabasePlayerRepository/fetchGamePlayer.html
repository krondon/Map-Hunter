Fetch game player data for a user in a specific event.

userId The user to query. eventId Optional event to filter by. gamePlayerId Optional specific gamePlayer ID to fetch. Returns the game_player data map, or null if not found.

Implementation

@override
Future<Map<String, dynamic>?> fetchGamePlayer({
  required String userId,
  String? eventId,
  String? gamePlayerId,
}) async {
  try {
    // If we have a specific gamePlayerId, fetch directly
    if (gamePlayerId != null) {
      return await _client
          .from('game_players')
          .select('id, lives, status, event_id')
          .eq('id', gamePlayerId)
          .maybeSingle();
    }

    // If we have an eventId, filter by it
    if (eventId != null) {
      return await _client
          .from('game_players')
          .select('id, lives, status, event_id')
          .eq('user_id', userId)
          .eq('event_id', eventId)
          .maybeSingle();
    }

    // Otherwise get the most recent
    return await _client
        .from('game_players')
        .select('id, lives, status, event_id')
        .eq('user_id', userId)
        .order('joined_at', ascending: false)
        .limit(1)
        .maybeSingle();
  } catch (e) {
    debugPrint('[PlayerRepo] Error fetching game player: $e');
    return null;
  }
}
