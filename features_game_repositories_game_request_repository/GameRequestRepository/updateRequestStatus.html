@override
Future<void> updateRequestStatus(String requestId, String status) async {
  try {
    await _supabase
        .from('game_requests')
        .update({'status': status})
        .eq('id', requestId);
  } catch (e) {
    debugPrint('GameRequestRepository: Error updating request: $e');
    rethrow;
  }
}
