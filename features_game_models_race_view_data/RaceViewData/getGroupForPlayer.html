ProgressGroup? getGroupForPlayer(String playerId) {
  final normalizedId = playerId.trim().toLowerCase();
  for (final group in progressGroups) {
    if (group.hasOverlap &&
        group.memberIds.any((id) => id.trim().toLowerCase() == normalizedId)) {
      return group;
    }
  }
  return null;
}
