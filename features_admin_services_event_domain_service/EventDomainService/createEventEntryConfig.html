static Map<String, dynamic> createEventEntryConfig({
  required EntryType entryType,
  double? entryFee,
  CurrencyType currency = CurrencyType.treboles,
}) {
  return {
    'entry_type': entryType.name,
    if (entryType == EntryType.paid) ...{
      'entry_fee': entryFee,
      'currency': currency.name,
    },
  };
}
